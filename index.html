function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheetName = workbook.SheetNames[0]; const sheet = workbook.Sheets[sheetName]; const jsonRows = XLSX.utils.sheet_to_json(sheet, { raw: false }); if(jsonRows.length === 0) { alert("File Excel trống!"); return; } let danhSachUpload = []; let setMaDonViExcel = new Set(); // Tính toán ID lớn nhất hiện tại trên Cloud bằng vòng lặp để tránh tràn bộ nhớ đệm let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { for (let i = 0; i < CACHE_NHAN_SU.length; i++) { let currentId = parseInt(CACHE_NHAN_SU[i].id) || 0; if (currentId > maxId) { maxId = currentId; } } } jsonRows.forEach(row => { let stt = "", hoTen = "", ten = "", soBhxh = "", soCccd = ""; let atv = "", nndh = "", namVeHuu = "", maDonVi = "", nam = "", namSinh = "", nu = "", chucDanh = ""; for (let key in row) { // Loại bỏ hoàn toàn mọi khoảng trắng, tab, xuống dòng trong tiêu đề cột let cleanKey = key.replace(/\s+/g, '').trim().toLowerCase(); // Quét từ khóa thông minh, linh hoạt loại trừ để tránh nhận diện nhầm lẫn if (cleanKey.includes("stt")) { stt = row[key]; } else if (cleanKey.includes("họvàtên") || cleanKey.includes("hoten") || (cleanKey.includes("họ") && cleanKey.includes("tên"))) { hoTen = row[key]; } else if (cleanKey === "tên" || cleanKey === "ten") { ten = row[key]; } else if (cleanKey.includes("bhxh") || cleanKey.includes("bảohiểm") || cleanKey.includes("sosổ")) { soBhxh = row[key]; } else if (cleanKey.includes("cccd") || cleanKey.includes("chứngminh") || cleanKey.includes("cmnd")) { soCccd = row[key]; } else if (cleanKey.includes("antoàn") || cleanKey.includes("atv")) { atv = row[key]; } else if (cleanKey.includes("nnđh") || cleanKey.includes("nndh")) { nndh = row[key]; } else if (cleanKey.includes("vềhưu") || cleanKey.includes("vehuu")) { namVeHuu = row[key]; } else if (cleanKey.includes("mãđơnvị") || cleanKey.includes("madonvi") || cleanKey.includes("đơnvị") || cleanKey.includes("mãđv")) { maDonVi = row[key]; } else if (cleanKey === "năm" || cleanKey === "nam") { nam = row[key]; } else if (cleanKey.includes("nămsinh") || cleanKey.includes("namsinh")) { namSinh = row[key]; } else if (cleanKey.includes("nữ") || cleanKey === "nu") { nu = row[key]; } else if (cleanKey.includes("chứcdanh") || cleanKey.includes("côngviệc") || cleanKey.includes("chucdanh")) { chucDanh = row[key]; } } // Nếu không quét được cột Họ và tên ghép, thử tìm bất kỳ cột nào có chữ "họ" làm phương án dự phòng if (!hoTen) { for (let k in row) { let kClean = k.replace(/\s+/g, '').trim().toLowerCase(); if (kClean.includes("họ") && !kClean.includes("hưu")) { hoTen = row[k]; break; } } } // Loại trừ tiêu đề hoặc các dòng rỗng, dòng chứa ký hiệu đánh dấu giới tính if (hoTen && String(hoTen).trim() !== "" && String(hoTen).trim().toLowerCase() !== "x") { let cleanCccd = String(soCccd || '').trim(); let cleanMaDonVi = String(maDonVi || '').replace(/\s+/g, '').trim(); if (cleanMaDonVi === "") { cleanMaDonVi = "DV-UNKNOWN"; } setMaDonViExcel.add(cleanMaDonVi); maxId++; danhSachUpload.push({ id: maxId, stt_file: String(stt || ''), ho_ten: String(hoTen).trim(), ten: String(ten || '').trim(), so_bhxh: String(soBhxh || ''), so_cccd: cleanCccd, an_toan_vien: String(atv || ''), nndh: String(nndh || ''), nam_ve_huu: formatExcelDate(namVeHuu), ma_don_vi: cleanMaDonVi, nam_khao_sat: formatExcelDate(nam), nam_sinh: formatExcelDate(namSinh), gioi_tinh_nu: String(nu || ''), chuc_danh_cong_viec: String(chucDanh || '') }); } }); if(danhSachUpload.length === 0) { alert("🚨 Hệ thống không tìm thấy dữ liệu hợp lệ! Hãy chắc chắn file đang nạp có chứa tiêu đề cột dạng 'Họ và tên'."); input.value = ""; return; } if (setMaDonViExcel.size > 0) { let mangMaDonVi = Array.from(setMaDonViExcel); let payloadDonVi = mangMaDonVi.map(m => ({ ma_don_vi: m, ten_don_vi: m === "DV-UNKNOWN" ? "Đơn vị chưa xác định" : `Đơn vị mã ${m}` })); const { error: errDonVi } = await client.from('don_vi').upsert(payloadDonVi, { onConflict: 'ma_don_vi' }); if(errDonVi) console.error("Lỗi đăng ký đơn vị:", errDonVi); } const { error: errNhanSu } = await client.from('nhan_su').insert(danhSachUpload); if(errNhanSu) { console.error("Lỗi Supabase:", errNhanSu); alert("🚨 Lỗi đẩy lên Cloud thất bại: \n" + errNhanSu.message); } else { alert(`🎉 Thành công hoàn hảo! Đã nhận diện và tải lên thành công ${danhSachUpload.length} nhân sự lên Cloud.`); } document.getElementById('search-ns').value = ""; input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi cấu trúc tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
