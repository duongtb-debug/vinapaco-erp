function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheetName = workbook.SheetNames[0]; const sheet = workbook.Sheets[sheetName]; // Chuyển sang mảng 2 chiều để định vị hàng chứa tiêu đề một cách chuẩn xác nhất const rows = XLSX.utils.sheet_to_json(sheet, { header: 1, raw: false }); if(rows.length === 0) { alert("File Excel trống!"); return; } let headerIdx = -1; let colMapping = { stt: -1, hoTen: -1, ten: -1, soBhxh: -1, soCccd: -1, atv: -1, nndh: -1, namVeHuu: -1, maDonVi: -1, nam: -1, namSinh: -1, nu: -1, chucDanh: -1 }; // Thuật toán quét thông minh: Tìm hàng thực sự chứa từ khóa "Họ và tên" for (let i = 0; i < rows.length; i++) { let row = rows[i]; if (!row) continue; for (let j = 0; j < row.length; j++) { if (!row[j]) continue; let val = String(row[j]).replace(/\s+/g, '').trim().toLowerCase(); if ((val.includes("họ") && val.includes("tên")) || val.includes("hoten") || val.includes("hovaten")) { colMapping.hoTen = j; } else if (val === "stt") { colMapping.stt = j; } else if (val === "tên" || val === "ten") { colMapping.ten = j; } else if (val.includes("bhxh") || val.includes("bảohiểm") || val.includes("sổbhxh")) { colMapping.soBhxh = j; } else if (val.includes("cccd") || val.includes("chứngminh") || val.includes("cmnd")) { colMapping.soCccd = j; } else if (val.includes("antoàn") || val.includes("atv")) { colMapping.atv = j; } else if (val.includes("nnđh") || val.includes("nndh")) { colMapping.nndh = j; } else if (val.includes("vềhưu") || val.includes("vehuu")) { colMapping.namVeHuu = j; } else if (val.includes("mãđơnvị") || val.includes("madonvi") || val.includes("mãđv")) { colMapping.maDonVi = j; } else if (val === "năm" || val === "nam" || val.includes("khảosát")) { colMapping.nam = j; } else if (val.includes("nămsinh") || val.includes("namsinh")) { colMapping.namSinh = j; } else if (val.includes("nữ") || val === "nu") { colMapping.nu = j; } else if (val.includes("chứcdanh") || val.includes("côngviệc") || val.includes("chucdanh")) { colMapping.chucDanh = j; } } // Nếu tìm thấy cột Họ và tên, xác lập đây là hàng tiêu đề và dừng quét if (colMapping.hoTen !== -1) { headerIdx = i; break; } } if (headerIdx === -1) { alert("🚨 Lỗi cấu trúc: Hệ thống không thể định vị cột 'Họ và tên' trong file Excel của bạn!"); input.value = ""; return; } let danhSachUpload = []; let setMaDonViExcel = new Set(); let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { for (let i = 0; i < CACHE_NHAN_SU.length; i++) { let currentId = parseInt(CACHE_NHAN_SU[i].id) || 0; if (currentId > maxId) { maxId = currentId; } } } // Đọc dữ liệu từ sau hàng tiêu đề trở đi for (let i = headerIdx + 1; i < rows.length; i++) { const row = rows[i]; if (!row || row.length === 0) continue; let hoTen = colMapping.hoTen !== -1 && row[colMapping.hoTen] ? String(row[colMapping.hoTen]).trim() : ""; if (!hoTen || hoTen.toLowerCase() === "họ và tên" || hoTen.toLowerCase() === "x") continue; let stt = colMapping.stt !== -1 && row[colMapping.stt] ? String(row[colMapping.stt]).trim() : ""; let ten = colMapping.ten !== -1 && row[colMapping.ten] ? String(row[colMapping.ten]).trim() : ""; let soBhxh = colMapping.soBhxh !== -1 && row[colMapping.soBhxh] ? String(row[colMapping.soBhxh]).trim() : ""; let soCccd = colMapping.soCccd !== -1 && row[colMapping.soCccd] ? String(row[colMapping.soCccd]).trim() : ""; let atv = colMapping.atv !== -1 && row[colMapping.atv] ? String(row[colMapping.atv]).trim() : ""; let nndh = colMapping.nndh !== -1 && row[colMapping.nndh] ? String(row[colMapping.nndh]).trim() : ""; let namVeHuu = colMapping.namVeHuu !== -1 && row[colMapping.namVeHuu] ? String(row[colMapping.namVeHuu]).trim() : ""; let maDonVi = colMapping.maDonVi !== -1 && row[colMapping.maDonVi] ? String(row[colMapping.maDonVi]).trim() : ""; let nam = colMapping.nam !== -1 && row[colMapping.nam] ? String(row[colMapping.nam]).trim() : ""; let namSinh = colMapping.namSinh !== -1 && row[colMapping.namSinh] ? String(row[colMapping.namSinh]).trim() : ""; let nu = colMapping.nu !== -1 && row[colMapping.nu] ? String(row[colMapping.nu]).trim() : ""; let chucDanh = colMapping.chucDanh !== -1 && row[colMapping.chucDanh] ? String(row[colMapping.chucDanh]).trim() : ""; let cleanMaDonVi = maDonVi.replace(/\s+/g, ''); if (cleanMaDonVi === "") { cleanMaDonVi = "DV-UNKNOWN"; } setMaDonViExcel.add(cleanMaDonVi); maxId++; danhSachUpload.push({ id: maxId, stt_file: stt, ho_ten: hoTen, ten: ten, so_bhxh: soBhxh, so_cccd: soCccd, an_toan_vien: atv, nndh: nndh, nam_ve_huu: formatExcelDate(namVeHuu), ma_don_vi: cleanMaDonVi, nam: formatExcelDate(nam), nam_sinh: formatExcelDate(namSinh), gioi_tinh_nu: nu, chuc_danh_cong_viec: chucDanh }); } if(danhSachUpload.length === 0) { alert("🚨 Không tìm thấy dữ liệu nhân sự hợp lệ!"); input.value = ""; return; } if (setMaDonViExcel.size > 0) { let mangMaDonVi = Array.from(setMaDonViExcel); let payloadDonVi = mangMaDonVi.map(m => ({ ma_don_vi: m, ten_don_vi: m === "DV-UNKNOWN" ? "Đơn vị chưa xác định" : `Đơn vị mã ${m}` })); await client.from('don_vi').upsert(payloadDonVi, { onConflict: 'ma_don_vi' }); } const { error: errNhanSu } = await client.from('nhan_su').insert(danhSachUpload); if(errNhanSu) { alert("🚨 Lỗi cơ sở dữ liệu khi tải lên Cloud: \n" + errNhanSu.message); } else { alert(`🎉 Thành công rực rỡ! Hệ thống đã nhận diện thông minh và nạp ${danhSachUpload.length} nhân sự lên hệ thống Cloud.`); } if(document.getElementById('search-ns')) document.getElementById('search-ns').value = ""; input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi cấu trúc tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
