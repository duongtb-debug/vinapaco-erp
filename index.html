function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheetName = workbook.SheetNames[0]; const sheet = workbook.Sheets[sheetName]; // Đọc toàn bộ sheet dưới dạng mảng hai chiều để quét động const rows = XLSX.utils.sheet_to_json(sheet, { header: 1, raw: false }); if(rows.length === 0) { alert("File Excel trống!"); return; } // BƯỚC 1: Tìm dòng tiêu đề (Header Row) và ánh xạ chỉ số cột (Index) let headerIdx = -1; let colMapping = { stt: -1, hoTen: -1, ten: -1, soBhxh: -1, soCccd: -1, atv: -1, nndh: -1, namVeHuu: -1, maDonVi: -1, nam: -1, namSinh: -1, nu: -1, chucDanh: -1 }; for (let i = 0; i < rows.length; i++) { let row = rows[i]; if (!row) continue; let matches = 0; for (let j = 0; j < row.length; j++) { if (!row[j]) continue; let val = String(row[j]).replace(/\s+/g, '').trim().toLowerCase(); if (val.includes("họ") && val.includes("tên")) { matches++; colMapping.hoTen = j; } else if (val === "stt") { colMapping.stt = j; } else if (val === "tên" || val === "ten") { colMapping.ten = j; } else if (val.includes("bhxh") || val.includes("bảohiểm")) { colMapping.soBhxh = j; } else if (val.includes("cccd") || val.includes("chứngminh") || val.includes("cmnd")) { colMapping.soCccd = j; } else if (val.includes("antoàn") || val.includes("atv")) { colMapping.atv = j; } else if (val.includes("nnđh") || val.includes("nndh")) { colMapping.nndh = j; } else if (val.includes("vềhưu") || val.includes("vehuu")) { colMapping.namVeHuu = j; } else if (val.includes("mãđơnvị") || val.includes("madonvi") || val.includes("mãđv") || (val.includes("mã") && val.includes("vị"))) { colMapping.maDonVi = j; } else if (val === "năm" || val === "nam") { colMapping.nam = j; } else if (val.includes("nămsinh") || val.includes("namsinh")) { colMapping.namSinh = j; } else if (val.includes("nữ") || val === "nu") { colMapping.nu = j; } else if (val.includes("chứcdanh") || val.includes("côngviệc")) { colMapping.chucDanh = j; } } // Nếu tìm thấy cột Họ và tên thì đây chính là dòng tiêu đề if (colMapping.hoTen !== -1) { headerIdx = i; break; } } if (headerIdx === -1) { alert("🚨 Lỗi cấu trúc: Hệ thống không thể tự động định vị được cột 'Họ và tên' trong file Excel của bạn!"); input.value = ""; return; } let danhSachUpload = []; let setMaDonViExcel = new Set(); // Tính toán ID an toàn tuyệt đối tránh trùng khóa chính dữ liệu cũ let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { for (let i = 0; i < CACHE_NHAN_SU.length; i++) { let currentId = parseInt(CACHE_NHAN_SU[i].id) || 0; if (currentId > maxId) { maxId = currentId; } } } // BƯỚC 2: Đọc dữ liệu từ sau dòng tiêu đề for (let i = headerIdx + 1; i < rows.length; i++) { const row = rows[i]; if (!row || row.length === 0) continue; let hoTen = colMapping.hoTen !== -1 && row[colMapping.hoTen] ? String(row[colMapping.hoTen]).trim() : ""; // Bỏ qua dòng trống hoặc dòng lặp lại chữ tiêu đề if (!hoTen || hoTen.toLowerCase() === "họ và tên" || hoTen.toLowerCase() === "x") continue; let stt = colMapping.stt !== -1 && row[colMapping.stt] ? String(row[colMapping.stt]).trim() : ""; let ten = colMapping.ten !== -1 && row[colMapping.ten] ? String(row[colMapping.ten]).trim() : ""; let soBhxh = colMapping.soBhxh !== -1 && row[colMapping.soBhxh] ? String(row[colMapping.soBhxh]).trim() : ""; let soCccd = colMapping.soCccd !== -1 && row[colMapping.soCccd] ? String(row[colMapping.soCccd]).trim() : ""; let atv = colMapping.atv !== -1 && row[colMapping.atv] ? String(row[colMapping.atv]).trim() : ""; let nndh = colMapping.nndh !== -1 && row[colMapping.nndh] ? String(row[colMapping.nndh]).trim() : ""; let namVeHuu = colMapping.namVeHuu !== -1 && row[colMapping.namVeHuu] ? String(row[colMapping.namVeHuu]).trim() : ""; let maDonVi = colMapping.maDonVi !== -1 && row[colMapping.maDonVi] ? String(row[colMapping.maDonVi]).trim() : ""; let nam = colMapping.nam !== -1 && row[colMapping.nam] ? String(row[colMapping.nam]).trim() : ""; let namSinh = colMapping.namSinh !== -1 && row[colMapping.namSinh] ? String(row[colMapping.namSinh]).trim() : ""; let nu = colMapping.nu !== -1 && row[colMapping.nu] ? String(row[colMapping.nu]).trim() : ""; let chucDanh = colMapping.chucDanh !== -1 && row[colMapping.chucDanh] ? String(row[colMapping.chucDanh]).trim() : ""; let cleanMaDonVi = maDonVi.replace(/\s+/g, ''); if (cleanMaDonVi === "") { cleanMaDonVi = "DV-UNKNOWN"; } setMaDonViExcel.add(cleanMaDonVi); maxId++; danhSachUpload.push({ id: maxId, stt_file: stt, ho_ten: hoTen, ten: ten, so_bhxh: soBhxh, so_cccd: soCccd, an_toan_vien: atv, nndh: nndh, nam_ve_huu: formatExcelDate(namVeHuu), ma_don_vi: cleanMaDonVi, nam_khao_sat: formatExcelDate(nam), nam_sinh: formatExcelDate(namSinh), gioi_tinh_nu: nu, chuc_danh_cong_viec: chucDanh }); } if(danhSachUpload.length === 0) { alert("🚨 Không tìm thấy danh sách nhân sự nào hợp lệ dưới dòng tiêu đề!"); input.value = ""; return; } if (setMaDonViExcel.size > 0) { let mangMaDonVi = Array.from(setMaDonViExcel); let payloadDonVi = mangMaDonVi.map(m => ({ ma_don_vi: m, ten_don_vi: m === "DV-UNKNOWN" ? "Đơn vị chưa xác định" : `Đơn vị mã ${m}` })); await client.from('don_vi').upsert(payloadDonVi, { onConflict: 'ma_don_vi' }); } const { error: errNhanSu } = await client.from('nhan_su').insert(danhSachUpload); if(errNhanSu) { alert("🚨 Lỗi đồng bộ đám mây: \n" + errNhanSu.message); } else { alert(`🎉 Xuất sắc! Đã nhận diện thông minh và nạp thành công ${danhSachUpload.length} nhân sự lên Cloud.`); } document.getElementById('search-ns').value = ""; input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi cấu trúc tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
