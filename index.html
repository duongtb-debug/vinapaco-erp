// Hàm tự động lấy dữ liệu từ Supabase và hiển thị lên giao diện web async function taiDuLieuNhanSu() { try { // 1. Gọi chính xác bảng nhan_su viết thường từ Supabase const { data, error } = await supabase .from('nhan_su') .select('*'); if (error) { console.error('Lỗi khi truy vấn Supabase:', error); alert('Không thể kết nối lấy dữ liệu, vui lòng kiểm tra lại cấu hình!'); return; } console.log('Dữ liệu tải về thành công:', data); // 2. Hiển thị tổng số lượng nhân sự lên thẻ đếm (Ví dụ: Thẻ có ID là total-employees) const theTongSo = document.getElementById('total-employees') || document.getElementById('totalEmployees'); if (theTongSo) { theTongSo.innerText = data.length; } // 3. Tìm thẻ tbody của bảng để đổ dữ liệu vào (Thay ID 'employee-table-body' bằng ID bảng của bạn nếu cần) const tableBody = document.getElementById('employee-table-body') || document.querySelector('table tbody'); if (!tableBody) { console.error('Không tìm thấy thẻ tbody của bảng trên giao diện HTML!'); return; } // Xóa sạch hàng dữ liệu cũ (hoặc chữ Loading...) trước khi chèn dữ liệu mới tableBody.innerHTML = ''; // 4. Duyệt qua từng nhân sự để vẽ thành các hàng (row) trên bảng web data.forEach((emp, index) => { const rowHTML = ` ${index + 1} ${emp.ho_ten || ''} ${emp.gioi_tinh || ''} ${emp.nam_sinh || ''} ${emp.chuc_danh_cong_viec || ''} ${emp.nam_ve_huu || 'Chưa rõ'} `; tableBody.insertAdjacentHTML('beforeend', rowHTML); }); } catch (err) { console.error('Đã xảy ra lỗi hệ thống:', err); } } // Kích hoạt hàm chạy ngay khi trang web tải xong giao diện document.addEventListener('DOMContentLoaded', taiDuLieuNhanSu);
