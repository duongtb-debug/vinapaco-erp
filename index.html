function processExcelNhansu(input) { const file = input.files[0]; if (!file) return; showStatus("⏳ Hệ thống đang bóc tách và xử lý chuẩn hóa ngày tháng Excel. Vui lòng đợi...", "loading"); const reader = new FileReader(); reader.onload = function(e) { try { const workbook = XLSX.read(new Uint8Array(e.target.result), { type: 'array', cellDates: true, // Ép thư viện tự động nhận diện định dạng Ngày/Tháng/Năm từ Excel dateNF: 'yyyy-mm-dd' }); const firstSheetName = workbook.SheetNames[0]; const sheetData = XLSX.utils.sheet_to_json(workbook.Sheets[firstSheetName], { header: 1, defval: "" }); if (sheetData.length === 0) { showStatus("❌ Thất bại: File Excel không có bất kỳ dữ liệu nào.", "error"); return; } let headerIdx = -1; let colIdxs = { name: -1, cccd: -1, atv: -1, nndh: -1, retire: -1, unit: -1, birth: -1, female: -1 }; let limitRows = Math.min(sheetData.length, 15); for (let i = 0; i < limitRows; i++) { let cells = sheetData[i]; let isHeader = false; for(let j = 0; j < cells.length; j++) { let cText = String(cells[j]).toLowerCase().replace(/[\s\n\r]/g, ''); if(cText.includes("họvàtên") || cText.includes("sốcccd") || cText.includes("nữ(") || cText.includes("nămsinh")) { isHeader = true; break; } } if (isHeader) { headerIdx = i; cells.forEach((cell, idx) => { let text = String(cell).toLowerCase().replace(/[\s\n\r]/g, ''); if (text.includes("họvàtên")) colIdxs.name = idx; else if (text.includes("cccd") || text.includes("chứngminh") || text.includes("sốcccd")) colIdxs.cccd = idx; else if (text.includes("antoànviên") || text.includes("atv")) colIdxs.atv = idx; else if (text.includes("nndh") || text.includes("độchại")) colIdxs.nndh = idx; else if (text.includes("vềhưu") || text.includes("hưutrí")) colIdxs.retire = idx; else if (text.includes("đơnvị") || text.includes("mãđơnvị")) colIdxs.unit = idx; else if (text.includes("nămsinh") || text.includes("ngàysinh")) colIdxs.birth = idx; else if (text.includes("nữ")) colIdxs.female = idx; }); break; } } if (headerIdx === -1) { headerIdx = 1; colIdxs = { name: 1, cccd: 4, atv: 6, nndh: 7, retire: 8, unit: 9, birth: 11, female: 12 }; } // HÀM ĐỔI SỐ EXCEL (46266...) THÀNH NGÀY THÁNG ĐẸP CHUẨN CHỮ function formatExcelDate(cellVal) { if (!cellVal) return "-"; // Nếu bản chất đã là Object Date của JS sau khi parse if (cellVal instanceof Date && !isNaN(cellVal)) { let y = cellVal.getFullYear(); let m = String(cellVal.getMonth() + 1).padStart(2, '0'); let d = String(cellVal.getDate()).padStart(2, '0'); return `${d}/${m}/${y}`; } // Nếu vẫn dính chuỗi số thô (ví dụ: 46266) let num = parseFloat(cellVal); if (!isNaN(num) && num > 2100) { let utc_days = Math.floor(num - 25569); let utc_value = utc_days * 86400; let date_info = new Date(utc_value * 1000); let y = date_info.getFullYear(); let m = String(date_info.getMonth() + 1).padStart(2, '0'); let d = String(date_info.getDate()).padStart(2, '0'); return `${d}/${m}/${y}`; } return String(cellVal).trim(); } let tempNhansu = []; for (let i = headerIdx + 1; i < sheetData.length; i++) { let row = sheetData[i]; if (!row || row.length === 0) continue; let nameVal = colIdxs.name !== -1 ? row[colIdxs.name] : row[1]; if (!nameVal || String(nameVal).trim() === "" || String(nameVal).includes("Họ và tên") || String(nameVal).includes("Stt")) continue; let cccdVal = colIdxs.cccd !== -1 ? row[colIdxs.cccd] : row[4]; let atvVal = colIdxs.atv !== -1 ? row[colIdxs.atv] : row[6]; let nndhVal = colIdxs.nndh !== -1 ? row[colIdxs.nndh] : row[7]; let unitVal = colIdxs.unit !== -1 ? row[colIdxs.unit] : row[9]; let femaleVal = colIdxs.female !== -1 ? row[colIdxs.female] : row[12]; // Xử lý thông minh cho cột Năm Sinh và Năm Về Hưu bằng hàm chuyển đổi chống lỗi số let rawBirth = colIdxs.birth !== -1 ? row[colIdxs.birth] : row[11]; let rawRetire = colIdxs.retire !== -1 ? row[colIdxs.retire] : row[8]; let cleanBirthStr = formatExcelDate(rawBirth); let cleanRetireStr = formatExcelDate(rawRetire); // Trích xuất lấy ra 4 số năm để tính toán tuổi chính xác let matchYear = cleanBirthStr.match(/\d{4}/); let by = matchYear ? parseInt(matchYear[0]) : 1980; let isFemale = String(femaleVal).trim().toUpperCase() === "X" || String(femaleVal).trim() === "1" || String(femaleVal).trim() === "NỮ"; tempNhansu.push({ stt: tempNhansu.length + 1, fullName: String(nameVal).trim(), cccd: String(cccdVal).trim(), atv: String(atvVal).trim(), nndh: String(nndhVal).trim(), retireTime: cleanRetireStr, unitId: String(unitVal).trim().toUpperCase() || "DV-01", birthOrigin: cleanBirthStr, gender: isFemale ? "Nữ" : "Nam", currentAge: (2026 - by) }); } if(tempNhansu.length === 0) { showStatus("❌ Lỗi cấu trúc: Định dạng các cột trong file không tương thích.", "error"); return; } dataNhansu = tempNhansu; localStorage.setItem('erp_precise_db', JSON.stringify(dataNhansu)); renderNhansu(); showStatus("🎉 ĐỒNG BỘ THÀNH CÔNG! Đã dịch thuật ngày tháng chuẩn và nạp " + dataNhansu.length + " hồ sơ vào hệ thống.", "success"); } catch(err) { console.error(err); showStatus("❌ Lỗi xử lý dữ liệu: " + err.message, "error"); } }; reader.readAsArrayBuffer(file); }
