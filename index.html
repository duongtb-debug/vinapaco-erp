function xuLyImportExcelNhanSu(input) { const file = input.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; const jsonData = XLSX.utils.sheet_to_json(worksheet); if (jsonData.length === 0) { alert("⚠️ File Excel không có dữ liệu hoặc trống!"); input.value = ""; return; } let danhSachUpload = []; let tuSinhMaId = CACHE_NHÂN_SỰ.length + 1; for(let i = 0; i < jsonData.length; i++) { const row = jsonData[i]; // Đọc chính xác theo key tiêu đề trong file Book1_chuan_hoa của bạn let hoTen = row["ho_va_ten"] || row["ho_ten"] || row["Họ và tên"]; let cccd = row["so_cccd"] || row["Số CCCD"] || ""; let chucVu = row["chuc_danh_cong_viec"] || row["chuc_vu"] || "Công nhân"; let ngaySinhRaw = row["ngay_sinh"] || row["nam_sinh"] || ""; let gioiTinh = row["nu_x"] === "X" || row["nu_x"] === "x" ? "Nữ" : "Nam"; let maDonVi = row["ma_don_vi"] || "Tổng công ty Giấy"; let soBhxh = row["so_bhxh"] || ""; // Định dạng lại ngày sinh nếu có dữ liệu ngày từ Excel (bỏ phần giờ 00:00:00 đi nếu có) let ngaySinh = ""; if (ngaySinhRaw) { ngaySinh = ngaySinhRaw.toString().split(' ')[0]; } // Nếu dòng này không có tên thì bỏ qua if (!hoTen) continue; // Tự động sinh Mã Nhân Sự liên tục dựa trên STT tăng dần let maNV = `NS-${String(tuSinhMaId).padStart(4, '0')}`; tuSinhMaId++; // Đóng gói đúng cấu trúc 14 trường để đẩy lên Supabase danhSachUpload.push({ ma_nhan_su: maNV, ho_ten: hoTen.toString().trim(), phong_ban: maDonVi.toString().trim(), // Gắn tạm mã đơn vị/phòng ban chuc_vu: chucVu.toString().trim(), gioi_tinh: gioiTinh, ngay_sinh: ngaySinh, noi_sinh: "", nguyen_quan: "", cccd: cccd.toString().trim(), ngay_cap: "", noi_cap: "", dia_chi_thuong_tru: "Theo Số BHXH: " + soBhxh, // Tận dụng lưu thêm thông tin cần thiết dia_chi_hien_tai: "Tổng công ty Giấy Việt Nam", trang_thai: "Đang làm việc" }); } if (danhSachUpload.length === 0) { alert("❌ Không quét được nhân sự nào hợp lệ. Vui lòng kiểm tra lại cột 'ho_va_ten' trong file."); input.value = ""; return; } if(confirm(`📦 Đã bóc tách thành công ${danhSachUpload.length} nhân sự từ file gốc.\nBấm OK để thực hiện Bulk Insert lên Cloud?`)) { // Đẩy toàn bộ mảng dữ liệu lên Supabase trong 1 request duy nhất const { data: insertedData, error } = await client.from('nhan_su').insert(danhSachUpload).select(); if (error) { console.error("Chi tiết lỗi Supabase:", error); alert("❌ Lỗi nạp dữ liệu lên đám mây: " + (error.message || JSON.stringify(error))); } else { alert(`🎉 Thành công! Đã nạp đồng bộ ${danhSachUpload.length} nhân sự vào hệ thống quản lý.`); if(insertedData) CACHE_NHÂN_SỰ = CACHE_NHÂN_SỰ.concat(insertedData); locGiaoDienNhanSu(); } } } catch(err) { console.error(err); alert("❌ Lỗi cấu trúc trong quá trình xử lý tệp!"); } input.value = ""; }; reader.readAsArrayBuffer(file); }
