CHỨC NĂNG: CHỌN FILE DATA EXCEL TẢI LÊN (Tự động bóc tách và Bulk Insert) document.getElementById('excel-file-input').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); const label = document.querySelector('.file-label'); // Đổi trạng thái giao diện khi đang xử lý file label.innerText = "⏳ ĐANG NẠP DỮ LIỆU..."; label.style.backgroundColor = "#e67e22"; reader.onload = async function(evt) { try { const data = evt.target.result; const workbook = XLSX.read(data, { type: 'binary' }); // Đọc dữ liệu từ Sheet đầu tiên const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; // Chuyển sang định dạng JSON const danhSachExcel = XLSX.utils.sheet_to_json(worksheet); if (danhSachExcel.length === 0) { alert("❌ File Excel trống hoặc không đúng cấu trúc!"); resetNutFile(); return; } // Đồng bộ cấu trúc bản đồ dữ liệu chuẩn (Mapping) const mangUpload = danhSachExcel.map(row => { let chucDanhGoc = row["Chức danh công việc"] || row["Chức danh"] || "Chưa rõ"; let bhxh = row["Số BHXH"] || row["BHXH"] || "---"; let cccd = row["Số CCCD"] || row["CCCD"] || "---"; let chuoiChiTietHopNhat = `${chucDanhGoc} || ${bhxh} || ${cccd}`; return { ho_ten: row["Họ và tên"] || row["Họ tên"] || "Ẩn danh", gioi_tinh: row["Giới tính"] || "---", nam_sinh: String(row["Năm sinh"] || row["Ngày sinh"] || "---"), nam_ve_huu: String(row["Năm nghỉ hưu"] || row["Ngày nghỉ hưu"] || "---"), ma_don_vi: String(row["Mã ĐV"] || row["Mã đơn vị"] || "---").trim(), chuc_danh_cong_viec: chuoiChiTietHopNhat }; }); // Gửi loạt dữ liệu lớn (Bulk Insert) lên Supabase const { error } = await client.from('nhan_su').insert(mangUpload); if (error) throw error; alert(`🚀 Tải lên thành công! Đã nạp mới ${mangUpload.length} nhân sự lên Cloud.`); lamMoiBangDuLieu(); // Làm mới lại bảng hiển thị giao diện } catch (err) { alert("Lỗi nạp dữ liệu Excel: " + err.message); console.error(err); } finally { resetNutFile(); } }; reader.readAsBinaryString(file); }); // Hàm hỗ trợ reset trạng thái nút chọn file Excel function resetNutFile() { const label = document.querySelector('.file-label'); label.innerText = "📥 CHỌN FILE DATA EXCEL TẢI LÊN"; label.style.backgroundColor = "#95a5a6"; document.getElementById('excel-file-input').value = ""; }
