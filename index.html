HÀM NẠP FILE EXCEL - ĐỒNG BỘ ĐA TÊN CỘT ĐỂ KHỚP VỚI CẢ DATABASE VÀ EXCEL function napExcelNhanSu(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; const rawRows = XLSX.utils.sheet_to_json(worksheet, { header: 1 }); if (!rawRows || rawRows.length === 0) { alert("File Excel trống rỗng!"); return; } let headerIndex = -1; let viTriTen = -1, viTriMa = -1, viTriGioiTinh = -1, viTriNamSinh = -1, viTriHuu = -1, viTriChiTiet = -1; // Dò dòng tiêu đề chuẩn xác for (let i = 0; i < rawRows.length; i++) { const currentRow = rawRows[i]; if (!currentRow || !Array.isArray(currentRow)) continue; for (let j = 0; j < currentRow.length; j++) { if (currentRow[j] === undefined || currentRow[j] === null) continue; let valStr = String(currentRow[j]).toLowerCase().trim().replace(/\s+/g, ''); if (valStr.includes('họtên') || valStr.includes('hoten') || valStr.includes('họvàtên') || valStr === 'tên') { headerIndex = i; viTriTen = j; } } if (headerIndex !== -1) { for (let j = 0; j < currentRow.length; j++) { if (currentRow[j] === undefined || currentRow[j] === null) continue; let valStr = String(currentRow[j]).toLowerCase().trim().replace(/\s+/g, ''); if (j === viTriTen) continue; if (valStr.includes('mã') || valStr.includes('madv') || valStr.includes('đơnvị')) viTriMa = j; if (valStr.includes('giới') || valStr.includes('gioi') || valStr.includes('nam/nữ')) viTriGioiTinh = j; if (valStr.includes('nămsinh') || valStr.includes('namsinh')) viTriNamSinh = j; if (valStr.includes('hưu') || valStr.includes('vềhưu')) viTriHuu = j; if (valStr.includes('chitiet') || valStr.includes('chứcdanh') || valStr.includes('việc') || valStr.includes('môtả')) viTriChiTiet = j; } break; } } if (headerIndex === -1 || viTriTen === -1) { headerIndex = 0; viTriTen = 0; viTriMa = 1; } const mangNhanSuMapping = []; for (let i = headerIndex + 1; i < rawRows.length; i++) { const row = rawRows[i]; if (!row || !Array.isArray(row) || row[viTriTen] === undefined || row[viTriTen] === null) continue; const ho_ten = String(row[viTriTen]).trim(); if (ho_ten === "" || ho_ten.toLowerCase().includes('họ tên')) continue; const gioi_tinh = viTriGioiTinh !== -1 && row[viTriGioiTinh] ? String(row[viTriGioiTinh]).trim() : '---'; const nam_sinh = viTriNamSinh !== -1 && row[viTriNamSinh] ? (parseInt(row[viTriNamSinh]) || null) : null; const nam_ve_huu = viTriHuu !== -1 && row[viTriHuu] ? (parseInt(row[viTriHuu]) || null) : null; const ma_don_vi = viTriMa !== -1 && row[viTriMa] ? String(row[viTriMa]).trim() : '---'; const chuc_danh_cong_viec = viTriChiTiet !== -1 && row[viTriChiTiet] ? String(row[viTriChiTiet]).trim() : '---'; // Lưu dữ liệu dưới cả 2 định dạng (tên viết thường database và tên có dấu) để chắc chắn Supabase nhận diện được cột mangNhanSuMapping.push({ ho_ten, "Họ & Tên": ho_ten, gioi_tinh, "Giới Tính": gioi_tinh, nam_sinh, "Năm Sinh": nam_sinh, nam_ve_huu, "Năm Nghỉ Hưu": nam_ve_huu, ma_don_vi, "Mã ĐV": ma_don_vi, chuc_danh_cong_viec, "Chi Tiết": chuc_danh_cong_viec }); } if(mangNhanSuMapping.length === 0) { alert("Không trích xuất được dữ liệu hợp lệ!"); return; } alert(`Đang đồng bộ ${mangNhanSuMapping.length} nhân sự lên đám mây...`); const { error } = await client.from('nhan_su').insert(mangNhanSuMapping); if(error) throw error; alert("🎉 Đồng bộ thành công 100%!"); taiDuLieuTuCloud(); } catch (err) { alert("Lỗi xử lý Excel: " + err.message); console.error(err); } }; reader.readAsArrayBuffer(file); input.value = ''; }
