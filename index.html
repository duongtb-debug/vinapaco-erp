function calculateNSKPIs() { let total = dataNhansu.length; if(total === 0) return; // 1. Tính toán số lượng và tỷ lệ Giới tính let femaleCount = dataNhansu.filter(e => e.gender === "Nữ").length; let maleCount = total - femaleCount; let femaleRatio = ((femaleCount / total) * 100).toFixed(1); // Làm tròn 1 chữ số thập phân let maleRatio = ((maleCount / total) * 100).toFixed(1); // 2. Tính toán số lượng và tỷ lệ Lao động độc hại let nndhCount = dataNhansu.filter(e => e.nndh && e.nndh !== "0" && e.nndh !== "").length; let nndhRatio = ((nndhCount / total) * 100).toFixed(1); // 3. Tính toán số lượng và tỷ lệ An toàn viên let atvCount = dataNhansu.filter(e => e.atv && e.atv !== "0" && e.atv !== "").length; let atvRatio = ((atvCount / total) * 100).toFixed(1); // 4. Tính toán độ tuổi trung bình let avgAge = Math.round(dataNhansu.reduce((sum, e) => sum + e.currentAge, 0) / total); // ĐẨY DỮ LIỆU HIỂN THỊ LÊN CÁC THẺ METRIC CARD TRÊN GIAO DIỆN CHÍNH document.getElementById('countTotal').innerHTML = `${total} người (Cơ sở gốc 100%)`; document.getElementById('countGenderRatio').innerHTML = ` 👨 Nam: ${maleCount} (${maleRatio}%)
👩 Nữ: ${femaleCount} (${femaleRatio}%) `; document.getElementById('countNndh').innerHTML = ` ${nndhCount} người
Tỷ lệ: ${nndhRatio}% `; document.getElementById('countAtv').innerHTML = ` ${atvCount} người
Tỷ lệ: ${atvRatio}% `; document.getElementById('avgAge').innerText = avgAge + " tuổi"; }
