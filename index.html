function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheetName = workbook.SheetNames[0]; const sheet = workbook.Sheets[sheetName]; const jsonRows = XLSX.utils.sheet_to_json(sheet, { raw: false }); if(jsonRows.length === 0) { alert("File Excel trống!"); return; } let danhSachUpload = []; let setMaDonViExcel = new Set(); // Tính toán ID an toàn tuyệt đối, dùng vòng lặp thông thường để tránh trắng màn hình let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { for (let i = 0; i < CACHE_NHAN_SU.length; i++) { let currentId = parseInt(CACHE_NHAN_SU[i].id) || 0; if (currentId > maxId) { maxId = currentId; } } } jsonRows.forEach(row => { let stt = "", hoTen = "", ten = "", soBhxh = "", soCccd = ""; let atv = "", nndh = "", namVeHuu = "", maDonVi = "", nam = "", namSinh = "", nu = "", chucDanh = ""; for (let key in row) { let cleanKey = key.replace(/\s+/g, '').trim().toLowerCase(); if (cleanKey === "stt") stt = row[key]; else if (cleanKey.includes("họ") || cleanKey.includes("hoten") || cleanKey.includes("ten")) hoTen = row[key]; else if (cleanKey === "tên" || cleanKey === "ten") ten = row[key]; else if (cleanKey.includes("bhxh") || cleanKey.includes("bảohiểm") || cleanKey.includes("so_so")) soBhxh = row[key]; else if (cleanKey.includes("cccd") || cleanKey.includes("chứngminh") || cleanKey.includes("cmnd")) soCccd = row[key]; else if (cleanKey.includes("antoànviên") || cleanKey.includes("antoan") || cleanKey.includes("atv")) atv = row[key]; else if (cleanKey === "nnđh" || cleanKey === "nndh") nndh = row[key]; else if (cleanKey.includes("vềhưu") || cleanKey.includes("vehuu")) namVeHuu = row[key]; else if (cleanKey.includes("mãđơnvị") || cleanKey.includes("madonvi") || cleanKey.includes("mã_đv")) maDonVi = row[key]; else if (cleanKey === "năm" || cleanKey === "nam") nam = row[key]; else if (cleanKey.includes("nămsinh") || cleanKey === "namsinh") namSinh = row[key]; else if (cleanKey.includes("nữ") || cleanKey === "nu") nu = row[key]; else if (cleanKey.includes("chứcdanh") || cleanKey.includes("côngviệc") || cleanKey.includes("chuc_danh")) chucDanh = row[key]; } // Nhận diện linh hoạt tiêu đề nếu hoTen trống nhưng ten có dữ liệu if (!hoTen && ten) hoTen = ten; if (hoTen && String(hoTen).trim() !== "") { let cleanCccd = String(soCccd || '').trim(); let cleanMaDonVi = String(maDonVi || '').replace(/\s+/g, '').trim(); if (cleanMaDonVi === "") { cleanMaDonVi = "DV-UNKNOWN"; } setMaDonViExcel.add(cleanMaDonVi); maxId++; // Luôn tự tăng ID để đảm bảo đẩy được dữ liệu lên mà không lo bị lọc trùng danhSachUpload.push({ id: maxId, stt_file: String(stt || ''), ho_ten: String(hoTen).trim(), ten: String(ten || '').trim(), so_bhxh: String(soBhxh || ''), so_cccd: cleanCccd, an_toan_vien: String(atv || ''), nndh: String(nndh || ''), nam_ve_huu: formatExcelDate(namVeHuu), ma_don_vi: cleanMaDonVi, nam_khao_sat: formatExcelDate(nam), nam_sinh: formatExcelDate(namSinh), gioi_tinh_nu: String(nu || ''), chuc_danh_cong_viec: String(chucDanh || '') }); } }); if(danhSachUpload.length === 0) { alert("🚨 Không tìm thấy dữ liệu hợp lệ! Vui lòng kiểm tra lại xem file Excel đã có dữ liệu ở cột 'Họ và tên' chưa."); input.value = ""; return; } if (setMaDonViExcel.size > 0) { let mangMaDonVi = Array.from(setMaDonViExcel); let payloadDonVi = mangMaDonVi.map(m => ({ ma_don_vi: m, ten_don_vi: m === "DV-UNKNOWN" ? "Đơn vị chưa xác định" : `Đơn vị mã ${m}` })); const { error: errDonVi } = await client.from('don_vi').upsert(payloadDonVi, { onConflict: 'ma_don_vi' }); if(errDonVi) console.error("Lỗi đăng ký đơn vị:", errDonVi); } // Thực hiện nạp toàn bộ danh sách vào cơ sở dữ liệu const { error: errNhanSu } = await client.from('nhan_su').insert(danhSachUpload); if(errNhanSu) { console.error("Lỗi Supabase:", errNhanSu); alert("🚨 Không thể nạp dữ liệu: \n" + errNhanSu.message); } else { alert(`🎉 Thành công! Hệ thống đã nạp toàn bộ ${danhSachUpload.length} nhân sự từ file Excel lên Cloud.`); } document.getElementById('search-ns').value = ""; input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi cấu trúc tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
