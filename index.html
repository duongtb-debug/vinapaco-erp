async function importExcelNhanSuRawArray(input) { if (!input.files || input.files.length === 0) return; const file = input.files[0]; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); // cellDates: true giúp tự động chuyển đổi định dạng ngày của Excel sang Date object trong JS const workbook = XLSX.read(data, { type: 'array', cellDates: true }); const sheet = workbook.Sheets[workbook.SheetNames[0]]; // Chuyển sheet thành mảng 2 chiều (Mảng các hàng) const rawRows = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: "" }); if (rawRows.length === 0) { alert("File Excel trống!"); return; } let headerRowIdx = -1; let headersCleaned = []; // BƯỚC 1: ĐỊNH VỊ CHÍNH XÁC DÒNG TIÊU ĐỀ (HEADER) for (let i = 0; i < rawRows.length; i++) { let rowArr = rawRows[i] || []; let cleaned = rowArr.map(cell => cleanStringForIndexMatch(cell)); // Tiêu chuẩn bắt buộc để nhận diện dòng tiêu đề: Phải có đồng thời cột 'họ tên' và 'cccd' (hoặc bhxh) // Điều này giúp loại bỏ hoàn toàn việc nhận nhầm dòng của cô "TRIỆU THỊ NĂM" làm tiêu đề. if ((cleaned.includes('hovaten') || cleaned.includes('hoten')) && cleaned.includes('socccd')) { headerRowIdx = i; headersCleaned = cleaned; break; } } if (headerRowIdx === -1) { alert("Không thể tìm thấy dòng tiêu đề chuẩn (Họ tên + Số CCCD) trong file Excel!"); return; } // BƯỚC 2: PHÂN TÁCH INDEX CỘT - KHÓA CHẶT TỪNG VỊ TRÍ let idxHoTen = headersCleaned.findIndex(h => h === 'hovaten' || h === 'hoten'); let idxTen = headersCleaned.findIndex(h => h === 'ten'); let idxBhxh = headersCleaned.findIndex(h => h.includes('bhxh') || h.includes('baohiem')); let idxCccd = headersCleaned.findIndex(h => h.includes('cccd') || h.includes('cmnd') || h === 'socccd'); let idxVeHuu = headersCleaned.findIndex(h => h.includes('vehuu')); let idxMaDv = headersCleaned.findIndex(h => h.includes('madonvi') || h.includes('donvi')); // TỐI ƯU CỐT LÕI: Tìm cột "Năm" bằng so sánh bằng tuyệt đối '===', không dùng 'includes' let idxNam = headersCleaned.findIndex(h => h === 'nam'); // Cột năm sinh tìm riêng biệt let idxNamSinh = headersCleaned.findIndex(h => h.includes('namsinh') || h === 'namsinh'); let idxNu = headersCleaned.findIndex(h => h.includes('nu')); let idxChucDanh = headersCleaned.findIndex(h => h.includes('chucdanh') || h.includes('congviec')); // BƯỚC 3: QUÉT DỮ LIỆU VÀ CHUẨN HÓA (ÉP KIỂU AN TOÀN) let bodyRows = rawRows.slice(headerRowIdx + 1); let dataParsed = bodyRows.map(row => { if (!row || row.length === 0) return null; let hoTenTxt = idxHoTen !== -1 && row[idxHoTen] ? String(row[idxHoTen]).trim() : ''; // Bỏ qua dòng trống hoặc dòng lặp lại tiêu đề if (!hoTenTxt || hoTenTxt.toLowerCase() === 'stt' || cleanStringForIndexMatch(hoTenTxt) === 'hovaten') return null; let tenRieng = idxTen !== -1 && row[idxTen] ? String(row[idxTen]).trim() : hoTenTxt.split(' ').pop(); let bhxh = idxBhxh !== -1 && row[idxBhxh] ? String(row[idxBhxh]).trim() : ''; let cccd = idxCccd !== -1 && row[idxCccd] ? String(row[idxCccd]).trim() : ''; let maDonVi = idxMaDv !== -1 && row[idxMaDv] !== undefined ? String(row[idxMaDv]).trim() : ''; // 🌟 TỐI ƯU XỬ LÝ CỘT NĂM (KIỂU INTEGER) let namGoc = 2026; // Năm mặc định nếu trống if (idxNam !== -1 && row[idxNam] !== undefined && row[idxNam] !== null && row[idxNam] !== "") { let rawNamVal = row[idxNam]; if (rawNamVal instanceof Date) { // Nếu Excel tự động biến ô này thành Date, chỉ bốc lấy năm (Ví dụ: 1969-08-09 -> lấy 1969) namGoc = rawNamVal.getFullYear(); } else { let strVal = String(rawNamVal).trim(); // Nếu là chuỗi ngày tháng dạng YYYY-MM-DD (chứa dấu gạch ngang) if (strVal.indexOf('-') !== -1) { let parts = strVal.split('-'); if (parts[0] && parts[0].length === 4) namGoc = parseInt(parts[0]); // Bốc 4 số đầu làm năm } else { // Nếu là chuỗi số thuần túy hoặc dính chữ, lọc chỉ lấy số let parsedNum = parseInt(strVal.replace(/[^0-9]/g, '')); if (!isNaN(parsedNum)) namGoc = parsedNum; } } } // 🌟 TỐI ƯU XỬ LÝ CỘT NĂM SINH (KIỂU DATE ISO: YYYY-MM-DD) let parseDateISO = (val) => { if (!val) return null; if (val instanceof Date) { // Trả về đúng định dạng YYYY-MM-DD mà Supabase yêu cầu return `${val.getFullYear()}-${String(val.getMonth() + 1).padStart(2, '0')}-${String(val.getDate()).padStart(2, '0')}`; } let s = String(val).trim(); if (s.match(/^\d{4}-\d{2}-\d{2}/)) return s.substring(0, 10); // Phòng hờ định dạng DD/MM/YYYY trong Excel chuỗi let parts = s.match(/^(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{4})/); if (parts) return `${parts[3]}-${parts[2].padStart(2,'0')}-${parts[1].padStart(2,'0')}`; return null; }; let namSinh = idxNamSinh !== -1 ? parseDateISO(row[idxNamSinh]) : null; let namVeHuu = idxVeHuu !== -1 ? parseDateISO(row[idxVeHuu]) : null; let nuVal = idxNu !== -1 && row[idxNu] ? cleanStringForIndexMatch(row[idxNu]) : ''; let gioiTinh = (nuVal === 'x' || nuVal === '1' || nuVal === 'true') ? "Nữ" : "Nam"; let chucDanh = idxChucDanh !== -1 && row[idxChucDanh] ? String(row[idxChucDanh]).trim() : ''; return { ho_va_ten: hoTenTxt, ten: tenRieng, so_so_bhxh: bhxh, so_cccd: cccd, nam_ve_hu: namVeHuu, ma_don_vi: maDonVi, nam: parseInt(namGoc), // CHẮC CHẮN LÀ SỐ NGUYÊN SẠCH nam_sinh: namSinh, // CHẮC CHẮN LÀ DATE HOẶC NULL gioi_tinh: gioiTinh, chuc_danh_cong_viec: chucDanh, an_toan_vien: false, nndh: false }; }).filter(x => x !== null); if (dataParsed.length === 0) { alert("Không lọc được dữ liệu nhân sự hợp lệ nào!"); return; } // ĐẨY LÊN SUPABASE const { error } = await client.from('nhan_su').insert(dataParsed); if (error) { alert("Lỗi Database: " + error.message); } else { alert(`🎉 Thành công! Đã đồng bộ an toàn ${dataParsed.length} hồ sơ nhân sự.`); await reloadAllSystemData(); } } catch (err) { alert("Lỗi Engine Excel: " + err.message); } }; reader.readAsArrayBuffer(file); } // Hàm chuẩn hóa chuỗi để so khớp cột tiêu đề function cleanStringForIndexMatch(val) { if (val === undefined || val === null) return ''; return String(val) .replace(/[\r\n]+/g, ' ') .toLowerCase() .normalize('NFD') .replace(/[\u0300-\u036f]/g, '') .replace(/[^a-z0-9]/g, ''); }
