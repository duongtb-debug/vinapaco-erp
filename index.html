function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheetName = workbook.SheetNames[0]; const sheet = workbook.Sheets[sheetName]; const jsonRows = XLSX.utils.sheet_to_json(sheet, { raw: false }); if(jsonRows.length === 0) { alert("File Excel trống!"); return; } let danhSachUpload = []; let setMaDonViExcel = new Set(); let danhSachCccdCu = CACHE_NHAN_SU.map(n => String(n.so_cccd || '').trim()); // Lấy ID lớn nhất hiện tại trên Cloud để tự tăng tuyến tính, tránh lỗi Not-Null và Unique let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { maxId = Math.max(...CACHE_NHAN_SU.map(n => parseInt(n.id) || 0)); } jsonRows.forEach(row => { let stt = "", hoTen = "", ten = "", soBhxh = "", soCccd = ""; let atv = "", nndh = "", namVeHuu = "", maDonVi = "", nam = "", namSinh = "", nu = "", chucDanh = ""; for (let key in row) { let cleanKey = key.replace(/\s+/g, '').trim().toLowerCase(); if (cleanKey === "stt") stt = row[key]; else if (cleanKey.includes("họvàtên") || cleanKey.includes("hoten")) hoTen = row[key]; else if (cleanKey === "tên" || cleanKey === "ten") ten = row[key]; else if (cleanKey.includes("bhxh") || cleanKey.includes("bảohiểm")) soBhxh = row[key]; else if (cleanKey.includes("cccd") || cleanKey.includes("chứngminh")) soCccd = row[key]; else if (cleanKey.includes("antoànviên") || cleanKey.includes("antoan")) atv = row[key]; else if (cleanKey === "nnđh" || cleanKey === "nndh") nndh = row[key]; else if (cleanKey.includes("vềhưu") || cleanKey.includes("vehuu")) namVeHuu = row[key]; else if (cleanKey.includes("mãđơnvị") || cleanKey.includes("madonvi")) maDonVi = row[key]; else if (cleanKey === "năm" || cleanKey === "nam") nam = row[key]; else if (cleanKey.includes("nămsinh") || cleanKey === "namsinh") namSinh = row[key]; else if (cleanKey.includes("nữ") || cleanKey === "nu") nu = row[key]; else if (cleanKey.includes("chứcdanh") || cleanKey.includes("côngviệc")) chucDanh = row[key]; } if (hoTen && String(hoTen).trim() !== "") { let cleanCccd = String(soCccd || '').trim(); let cleanMaDonVi = String(maDonVi || '').replace(/\s+/g, '').trim(); if (cleanMaDonVi === "") { cleanMaDonVi = "DV-UNKNOWN"; } setMaDonViExcel.add(cleanMaDonVi); if (cleanCccd === "" || !danhSachCccdCu.includes(cleanCccd)) { maxId++; // Tự cộng lũy tiến ID để luôn tạo ra khóa chính hợp lệ danhSachUpload.push({ id: maxId, // Truyền trực tiếp ID số nguyên hợp lệ lên Supabase stt_file: String(stt || ''), ho_ten: String(hoTen).trim(), ten: String(ten || '').trim(), so_bhxh: String(soBhxh || ''), so_cccd: cleanCccd, an_toan_vien: String(atv || ''), nndh: String(nndh || ''), nam_ve_huu: formatExcelDate(namVeHuu), ma_don_vi: cleanMaDonVi, nam_khao_sat: formatExcelDate(nam), nam_sinh: formatExcelDate(namSinh), gioi_tinh_nu: String(nu || ''), chuc_danh_cong_viec: String(chucDanh || '') }); } } }); if(danhSachUpload.length === 0) { alert("Không có dữ liệu mới hoặc tất cả CCCD đã tồn tại trên Cloud!"); input.value = ""; return; } if (setMaDonViExcel.size > 0) { let mangMaDonVi = Array.from(setMaDonViExcel); let payloadDonVi = mangMaDonVi.map(m => ({ ma_don_vi: m, ten_don_vi: m === "DV-UNKNOWN" ? "Đơn vị chưa xác định" : `Đơn vị mã ${m}` })); const { error: errDonVi } = await client.from('don_vi').upsert(payloadDonVi, { onConflict: 'ma_don_vi' }); if(errDonVi) console.error("Lỗi đăng ký đơn vị:", errDonVi); } const { error: errNhanSu } = await client.from('nhan_su').insert(danhSachUpload); if(errNhanSu) { console.error("Lỗi Supabase:", errNhanSu); alert("🚨 Không thể nạp dữ liệu: \n" + errNhanSu.message); } else { alert(`🎉 Thành công dứt điểm! Hệ thống đã nạp ${danhSachUpload.length} nhân sự mới lên Cloud thành công!`); } document.getElementById('search-ns').value = ""; input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi cấu trúc tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
