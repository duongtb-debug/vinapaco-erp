function napExcelNhanSu(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; // ĐỌC THÔ TOÀN BỘ FILE THEO DẠNG MẢNG (Bao gồm cả dòng trống, gộp ô) const rawRows = XLSX.utils.sheet_to_json(worksheet, { header: 1 }); if (!rawRows || rawRows.length === 0) { alert("File Excel này hoàn toàn trống rỗng!"); return; } // BƯỚC 1: Đi tìm dòng chứa Tiêu Đề (Header) thực sự của file let headerIndex = -1; let viTriTen = -1, viTriMa = -1, viTriGioiTinh = -1, viTriNamSinh = -1, viTriHuu = -1, viTriChiTiet = -1; for (let i = 0; i < rawRows.length; i++) { const row = rawRows[i].map(cell => cell ? String(cell).toLowerCase().trim() : ''); // Dò xem dòng này có chứa từ khóa của cột Họ Tên không const indexTen = row.findIndex(c => c.includes('họ') && (c.includes('tên') || c.includes('ten'))); const indexMa = row.findIndex(c => c.includes('mã') || c.includes('ma đv') || c.includes('đơn vị')); if (indexTen !== -1) { headerIndex = i; // Đã tìm thấy dòng tiêu đề gốc! viTriTen = indexTen; viTriMa = indexMa; // Dò tiếp các vị trí cột còn lại trên chính dòng này viTriGioiTinh = row.findIndex(c => c.includes('giới tính') || c.includes('gioi')); viTriNamSinh = row.findIndex(c => c.includes('năm sinh') || c.includes('nam sinh')); viTriHuu = row.findIndex(c => c.includes('hưu') || c.includes('về hưu')); viTriChiTiet = row.findIndex(c => c.includes('chi tiết') || c.includes('chức danh') || c.includes('việc')); break; } } if (headerIndex === -1 || viTriTen === -1) { alert("❌ Không thể tìm thấy cột dữ liệu 'Họ & Tên' trong file Excel này. Vui lòng kiểm tra lại file!"); return; } // BƯỚC 2: Hốt sạch dữ liệu từ các dòng nằm DƯỚI dòng tiêu đề const mangNhanSuMapping = []; for (let i = headerIndex + 1; i < rawRows.length; i++) { const row = rawRows[i]; if (!row || !row[viTriTen]) continue; // Bỏ qua nếu dòng rỗng hoặc không có tên const ho_ten = String(row[viTriTen]).trim(); const gioi_tinh = viTriGioiTinh !== -1 && row[viTriGioiTinh] ? String(row[viTriGioiTinh]).trim() : ''; const nam_sinh = viTriNamSinh !== -1 && row[viTriNamSinh] ? parseInt(row[viTriNamSinh]) || null : null; const nam_ve_huu = viTriHuu !== -1 && row[viTriHuu] ? parseInt(row[viTriHuu]) || null : null; const ma_don_vi = viTriMa !== -1 && row[viTriMa] ? String(row[viTriMa]).trim() : ''; const chuc_danh_cong_viec = viTriChiTiet !== -1 && row[viTriChiTiet] ? String(row[viTriChiTiet]).trim() : ''; mangNhanSuMapping.push({ ho_ten, gioi_tinh, nam_sinh, nam_ve_huu, ma_don_vi, chuc_danh_cong_viec }); } if(mangNhanSuMapping.length === 0) { alert("❌ Không trích xuất được nhân sự nào hợp lệ dưới dòng tiêu đề!"); return; } alert(`Đang tiến hành đẩy dữ liệu ${mangNhanSuMapping.length} nhân sự lên đám mây...`); const { error } = await client.from('nhan_su').insert(mangNhanSuMapping); if(error) throw error; alert("🎉 Nạp dữ liệu Excel lên Cloud thành công 100%!"); taiDuLieuTuCloud(); } catch (err) { alert("Lỗi xử lý file Excel: " + err.message); console.error(err); } }; reader.readAsArrayBuffer(file); // ĐÂY LÀ ĐIỂM QUAN TRỌNG: Xóa trắng giá trị ô nhập file để nạp lại chính file đó không bị trơ input.value = ''; }
