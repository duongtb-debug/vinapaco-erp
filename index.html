Logic nút "XÓA SẠCH MÂY ĐỂ UP MỚI" document.getElementById('btn-clear-cloud').addEventListener('click', async () => { if (!confirm("⚠️ CẢNH BÁO: Hành động này sẽ xóa TOÀN BỘ dữ liệu nhân sự trên Cloud của Tổng công ty! Bạn có chắc chắn muốn xóa không?")) return; try { document.getElementById('btn-clear-cloud').innerText = "⏳ ĐANG XÓA..."; document.getElementById('btn-clear-cloud').disabled = true; // Xóa toàn bộ hàng bằng cách sử dụng filter không rỗng (ví dụ: id lớn hơn 0) const { error } = await client.from('nhan_su').delete().gt('id', 0); if (error) throw error; alert("🎉 Đã xóa sạch dữ liệu trên đám mây!"); lamMoiBangDuLieu(); // Cập nhật lại giao diện bảng trống } catch (e) { alert("Lỗi khi xóa dữ liệu: " + e.message); } finally { document.getElementById('btn-clear-cloud').innerText = "🗑️ XÓA SẠCH MÂY ĐỂ UP MỚI"; document.getElementById('btn-clear-cloud').disabled = false; } }); // 2. Logic đọc và tải File Excel lên Đám mây tự động document.getElementById('excel-file-input').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = async function(evt) { try { const data = evt.target.result; const workbook = XLSX.read(data, { type: 'binary' }); // Lấy sheet đầu tiên của file Excel const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; // Chuyển đổi dữ liệu sang định dạng JSON mảng danh sách const danhSachExcel = XLSX.utils.sheet_to_json(worksheet); if (danhSachExcel.length === 0) { alert("File Excel trống hoặc không đúng định dạng!"); return; } alert(`Tìm thấy ${danhSachExcel.length} nhân sự trong file. Tiến hành đồng bộ lên Cloud...`); // Khớp các cột từ file Excel của bạn tương ứng với các cột trong DB Supabase const mangUpload = danhSachExcel.map(row => { // Xử lý ghép cột chức danh, BHXH, CCCD thành định dạng chuỗi "||" của hệ thống nếu cần let chucDanhGoc = row["Chức danh công việc"] || row["Chức danh"] || ""; let bhxh = row["Số BHXH"] || row["BHXH"] || "---"; let cccd = row["Số CCCD"] || row["CCCD"] || "---"; let chuoiChucDanhHopNhat = `${chucDanhGoc} || ${bhxh} || ${cccd}`; return { ho_ten: row["Họ và tên"] || row["Họ tên"] || "Không tên", nam_sinh: row["Ngày sinh"] || row["Năm sinh"] || "---", nam_ve_huu: row["Ngày nghỉ hưu"] || row["Năm nghỉ hưu"] || "---", gioi_tinh: row["Giới tính"] || "---", ma_don_vi: String(row["Mã ĐV"] || row["Mã đơn vị"] || "---").trim(), chuc_danh_cong_viec: chuoiChucDanhHopNhat }; }); // Gửi dữ liệu loạt lớn (Bulk Insert) lên Supabase const { error } = await client.from('nhan_su').insert(mangUpload); if (error) throw error; alert("🚀 Tải lên và nạp dữ liệu mới thành công!"); document.getElementById('excel-file-input').value = ""; // Reset input file lamMoiBangDuLieu(); // Reload lại bảng } catch (err) { alert("Lỗi nạp dữ liệu file Excel: " + err.message); console.error(err); } }; reader.readAsBinaryString(file); });
