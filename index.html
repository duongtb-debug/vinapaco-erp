function locGiaoDienThanhToan() { const tbody = document.getElementById('body-tt-table'); if (!tbody) return; tbody.innerHTML = ''; let tongTienDaTT = 0; let tongTienChoDuyet = 0; const keyword = document.getElementById('search-tt') ? document.getElementById('search-tt').value.toLowerCase().trim() : ""; CACHE_THANH_TOÁN.forEach((tt, idx) => { // Kiểm tra điều kiện tìm kiếm (nếu có) if (keyword && !String(tt.ma_ho_so || '').toLowerCase().includes(keyword) && !String(tt.ma_thiet_bi || '').toLowerCase().includes(keyword) && !String(tt.don_vi_sua_chua || '').toLowerCase().includes(keyword)) { return; } // GIẢI PHÁP DỨT ĐIỂM: Ép kiểu an toàn, nếu null/undefined sẽ tự động lấy giá trị là 0 let soTienNum = Number(tt.so_tien) || 0; if (tt.trang_thai === "Đã thanh toán") { tongTienDaTT += soTienNum; } else { tongTienChoDuyet += soTienNum; } let badgeClass = tt.trang_thai === "Đã thanh toán" ? "status-good" : "status-warn"; let fileBtn = tt.file_data ? `` : `❌ Không có`; tbody.innerHTML += ` ${tt.ma_ho_so || '---'} ${tt.ma_thiet_bi || '---'} ${tt.don_vi_sua_chua || "---"} ${soTienNum.toLocaleString('vi-VN')} Đ ${tt.trang_thai || ''} ${fileBtn} `; }); // Cập nhật lại các thẻ thống kê ngoài giao diện an toàn if (document.getElementById('stat-tong-hs')) document.getElementById('stat-tong-hs').innerText = CACHE_THANH_TOÁN.length; if (document.getElementById('stat-tien-da-tt')) document.getElementById('stat-tien-da-tt').innerText = tongTienDaTT.toLocaleString('vi-VN') + " VNĐ"; if (document.getElementById('stat-tien-cho-duyet')) document.getElementById('stat-tien-cho-duyet').innerText = tongTienChoDuyet.toLocaleString('vi-VN') + " VNĐ"; }
