function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheet = workbook.Sheets[workbook.SheetNames[0]]; const rows = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: "" }); if (rows.length === 0) { alert("File Excel trống!"); return; } // Hàm xóa dấu tiếng Việt và ký tự đặc biệt để so khớp chính xác tuyệt đối function chuanHoaKey(k) { if (!k) return ""; let str = k.toString().toLowerCase().trim(); str = str.replace(/à|á|ạ|ả|ã|â|ầ|ấ|ậ|ẩ|ẫ|ă|ằ|ắ|ặ|ẳ|ẵ/g, "a"); str = str.replace(/è|é|ẹ|ẻ|ẽ|ê|ề|ế|ệ|ể|ễ/g, "e"); str = str.replace(/ì|í|ị|ỉ|ĩ/g, "i"); str = str.replace(/ò|ó|ọ|ỏ|õ|ô|ồ|ố|ộ|ổ|ỗ|ơ|ờ|ớ|ợ|ở|ỡ/g, "o"); str = str.replace(/ù|ú|ụ|ủ|ũ|ư|ừ|ứ|ự|ử|ữ/g, "u"); str = str.replace(/ỳ|ý|ỵ|ỷ|ỹ/g, "y"); str = str.replace(/đ/g, "d"); return str.replace(/\s+/g, '').replace(/[^a-z0-9]/g, ''); } let headerIdx = -1; let colMapping = { hoten: -1, cccd: -1, bhxh: -1, madonvi: -1, chucdanh: -1, namsinh: -1 }; // Nâng số dòng quét tìm tiêu đề từ 10 lên 30 dòng để tránh file có header quá dài for (let i = 0; i < Math.min(rows.length, 30); i++) { let row = rows[i]; if (!row) continue; let foundHeader = false; row.forEach((cell) => { let text = chuanHoaKey(cell); // Mở rộng bộ từ khóa nhận diện cột Họ Tên if (text.includes("hoten") || text.includes("hovaten") || text.includes("tennhanvien") || text.includes("canbo") || text === "ten") { foundHeader = true; } }); if (foundHeader) { headerIdx = i; row.forEach((cell, cellIdx) => { let text = chuanHoaKey(cell); if (text.includes("hoten") || text.includes("hovaten") || text.includes("tennhanvien") || text === "ten") colMapping.hoten = cellIdx; else if (text.includes("cccd") || text.includes("cmnd") || text.includes("chungminh") || text.includes("sochuyenduong")) colMapping.cccd = cellIdx; else if (text.includes("bhxh") || text.includes("baohiem") || text.includes("sosobhxh")) colMapping.bhxh = cellIdx; else if (text.includes("madonvi") || text.includes("bophan") || text.includes("phongban") || text.includes("donvi")) colMapping.madonvi = cellIdx; else if (text.includes("chucdanh") || text.includes("congviec") || text.includes("vitri") || text.includes("chucvu")) colMapping.chucdanh = cellIdx; else if (text.includes("namsinh") || text.includes("ngaysinh") || text === "nam") colMapping.namsinh = cellIdx; }); break; } } // BẪY LỖI DỰ PHÒNG: Nếu quét thông minh vẫn trượt, ép buộc lấy luôn Cột số 1 (Cột B) hoặc Cột số 0 (Cột A) làm Họ Tên if (headerIdx === -1 || colMapping.hoten === -1) { console.log("Không tìm thấy tiêu đề chuẩn, kích hoạt chế độ đọc ép buộc cột..."); headerIdx = 0; // Giả định dữ liệu bắt đầu từ dòng 1 hoặc 2 colMapping.hoten = 1; // Mặc định cột 2 (Cột B) là Họ tên colMapping.cccd = 2; // Mặc định cột C là CCCD colMapping.bhxh = 3; // Mặc định cột D là BHXH } let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { CACHE_NHAN_SU.forEach(ns => { if (parseInt(ns.id) > maxId) maxId = parseInt(ns.id); }); } let recordsToUpload = []; for (let i = headerIdx + 1; i < rows.length; i++) { let row = rows[i]; if (!row || row.length === 0) continue; let hoTenVal = String(row[colMapping.hoten] || "").trim(); // Bỏ qua dòng trống hoặc dòng lặp lại chữ tiêu đề if (!hoTenVal || hoTenVal.toLowerCase() === "họ và tên" || hoTenVal.toLowerCase() === "ho ten") continue; let cccdVal = colMapping.cccd !== -1 ? String(row[colMapping.cccd] || "").trim() : null; let bhxhVal = colMapping.bhxh !== -1 ? String(row[colMapping.bhxh] || "").trim() : null; let maDVVal = colMapping.madonvi !== -1 ? String(row[colMapping.madonvi] || "").trim().replace(/\s+/g, '') : "DV-01"; let chucDanhVal = colMapping.chucdanh !== -1 ? String(row[colMapping.chucdanh] || "").trim() : "Nhân viên"; let namSinhRaw = colMapping.namsinh !== -1 ? String(row[colMapping.namsinh] || "").trim() : ""; let namSinhVal = (typeof formatExcelDate === "function") ? formatExcelDate(namSinhRaw) : namSinhRaw; maxId++; recordsToUpload.push({ id: maxId, ho_ten: hoTenVal, so_cccd: cccdVal, so_bhxh: bhxhVal, ma_don_vi: maDVVal, chuc_danh_cong_viec: chucDanhVal, nam_sinh: namSinhVal, nam: namSinhVal // Điền thêm vào trường 'nam' phòng trường hợp cấu trúc cũ yêu cầu }); } if (recordsToUpload.length === 0) { alert("Không lấy được dữ liệu nhân sự nào hợp lệ từ file Excel này!"); input.value = ""; return; } // Tiến hành gửi lên Supabase const { error } = await client.from('nhan_su').insert(recordsToUpload); if (error) { alert("Lỗi tải lên Cloud: " + error.message); } else { alert("Thành công! Đã nạp thành công " + recordsToUpload.length + " nhân sự lên hệ thống."); } input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi xử lý tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
