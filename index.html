// ========================================================================= // THUẬT TOÁN ĐỊNH VỊ CHUẨN XÁC V5.2 - KHÓA TIÊU ĐỀ TUYỆT ĐỐI KHÔNG LO LỆCH CỘT // ========================================================================= function xuLyNhapFileExcelCloudSieuToc(event) { let file = event.target.files[0]; if(!file) return; let reader = new FileReader(); reader.onload = function(e) { try { let dataArray = new Uint8Array(e.target.result); let workbook = XLSX.read(dataArray, {type: 'array'}); let sheetRows = XLSX.utils.sheet_to_json(workbook.Sheets[workbook.SheetNames[0]], {header: 1, defval: ""}); if (sheetRows.length === 0) { alert("File Excel rỗng!"); return; } // BƯỚC 1: Tìm dòng tiêu đề (Header Row) let idxH = -1; for(let i = 0; i < Math.min(25, sheetRows.length); i++) { let rowText = sheetRows[i].map(c => c.toString().trim().toUpperCase()).join(" "); if(rowText.includes("HỌ VÀ TÊN") || rowText.includes("HỌ TÊN") || rowText.includes("HO TEN")) { idxH = i; break; } } if (idxH === -1) { alert("Không tìm thấy dòng tiêu đề chứa cột 'Họ và tên'!"); return; } // BƯỚC 2: Định vị chính xác Chỉ mục (Index) của từng cột dựa trên tiêu đề let headerRow = sheetRows[idxH].map(c => c.toString().trim().toUpperCase()); let colIndices = { hoTen: headerRow.findIndex(c => c.includes("HỌ VÀ TÊN") || c === "HỌ TÊN"), nam: headerRow.findIndex(c => c === "NĂM"), ngaySinh: headerRow.findIndex(c => c.includes("NGÀY SINH")), nghiHuu: headerRow.findIndex(c => c.includes("NGHỈ HƯU")), gioiTinh: headerRow.findIndex(c => c.includes("GIỚI TÍNH")), maDonVi: headerRow.findIndex(c => c.includes("MÃ ĐƠN VỊ") || c.includes("MÃ ĐV")), chucDanh: headerRow.findIndex(c => c.includes("CHỨC DANH") || c.includes("CÔNG VIỆC")), bhxh: headerRow.findIndex(c => c.includes("BHXH") || c.includes("BẢO HIỂM")), cccd: headerRow.findIndex(c => c.includes("CCCD") || c.includes("CMND")), atv: headerRow.findIndex(c => c.includes("ATV") || c.includes("AN TOÀN VIÊN")), docHai: headerRow.findIndex(c => c.includes("ĐỘC HẠI") || c.includes("NNDH") || c.includes("NẶNG NHỌC")) }; // Dự phòng nếu không tìm thấy một số cột phụ bằng tiêu đề chính (Quét ma trận tầng kép) for (let i = 0; i <= idxH; i++) { sheetRows[i].forEach((cellText, cellIndex) => { let textClean = cellText.toString().trim().toUpperCase(); if (colIndices.atv === -1 && (textClean.includes("ATV") || textClean.includes("AN TOÀN VIÊN"))) colIndices.atv = cellIndex; if (colIndices.docHai === -1 && (textClean.includes("ĐỘC HẠI") || textClean.includes("NNDH"))) colIndices.docHai = cellIndex; }); } let goiDuLieuDauRa = {}; let count = 0; // BƯỚC 3: Trích xuất dữ liệu chuẩn hóa theo sơ đồ cấu trúc đã khóa for(let i = idxH + 1; i < sheetRows.length; i++) { let r = sheetRows[i]; // Bỏ qua dòng trống hoặc dòng tổng kết cuối trang if(!r || r.length === 0) continue; let tenNhanSu = colIndices.hoTen !== -1 ? r[colIndices.hoTen] : r[1]; if(!tenNhanSu || tenNhanSu.toString().trim() === "" || tenNhanSu.toString().includes("CỘNG") || tenNhanSu.toString().includes("Tổng cộng")) continue; // Hàm trích xuất an toàn theo Index quy định let getVal = (index) => (index !== -1 && r[index] !== undefined) ? r[index].toString().trim() : "-"; let HoTen = tenNhanSu.toString().trim().toUpperCase(); let Nam = getVal(colIndices.nam) !== "-" ? getVal(colIndices.nam) : "2026"; let NamSinh = dinhDangNgayThangChuanExcel(getVal(colIndices.ngaySinh)); let NamNghiHuu = dinhDangNgayThangChuanExcel(getVal(colIndices.nghiHuu)); let GioiTinh = getVal(colIndices.gioiTinh); let MaDonVi = getVal(colIndices.maDonVi) !== "-" ? getVal(colIndices.maDonVi) : "CHƯA XẾP"; let ChucDanh = getVal(colIndices.chucDanh); let BHXH = getVal(colIndices.bhxh); let CCCD = getVal(colIndices.cccd); // Xử lý chuẩn hóa nhãn cột đặc thù (ATV và Độc hại) let ATV = "-"; let valATV = getVal(colIndices.atv).toLowerCase(); if(valATV === "1" || valATV === "x" || valATV === "✓" || valATV.includes("có")) ATV = "1"; let NNDH = "-"; let valDH = getVal(colIndices.docHai).toLowerCase(); if(valDH === "1" || valDH === "x" || valDH === "✓" || valDH.includes("có")) NNDH = "1"; // Đóng gói cấu trúc JSON đẩy lên Firebase let item = { HoTen, Nam, NamSinh, NamNghiHuu, GioiTinh, MaDonVi, ChucDanh, BHXH, CCCD, ATV, NNDH }; let keyAuto = firebaseDB.ref().child('nhansu').push().key; goiDuLieuDauRa[keyAuto] = item; count++; } if(count > 0) { firebaseDB.ref("nhansu").set(goiDuLieuDauRa, (err) => { if (!err) alert(`Đồng bộ v5.2 thành công! Đã bóc tách chuẩn xác ${count} nhân sự lên trục Cloud Singapore.`); }); } else { alert("Không bóc tách được dòng nhân sự nào, vui lòng kiểm tra lại file Excel mẫu!"); } } catch(err) { alert("Lỗi xử lý file hệ thống: " + err.message); } }; reader.readAsArrayBuffer(file); }
