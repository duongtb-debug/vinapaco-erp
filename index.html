function docVaUploadNhanSuTuExcel(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const sheetName = workbook.SheetNames[0]; const sheet = workbook.Sheets[sheetName]; // Đọc ép về dạng mảng gồm các mảng con (Array of Arrays) để cố định chỉ số cột chuẩn 100% const rows = XLSX.utils.sheet_to_json(sheet, { header: 1, raw: false }); if(rows.length <= 1) { alert("File Excel trống hoặc chỉ có tiêu đề!"); return; } let danhSachUpload = []; let setMaDonViExcel = new Set(); // Tính toán ID lớn nhất hiện tại trên Cloud bằng vòng lặp tránh sập màn hình let maxId = 0; if (CACHE_NHAN_SU && CACHE_NHAN_SU.length > 0) { for (let i = 0; i < CACHE_NHAN_SU.length; i++) { let currentId = parseInt(CACHE_NHAN_SU[i].id) || 0; if (currentId > maxId) { maxId = currentId; } } } // Duyệt từ dòng thứ 2 (Index 1) trở đi để bỏ qua dòng tiêu đề gốc for (let i = 1; i < rows.length; i++) { const row = rows[i]; if (!row || row.length === 0) continue; // Định vị chuẩn xác 100% dựa vào vị trí cột trong file Excel thực tế let stt = row[0] ? String(row[0]).trim() : ""; let hoTen = row[1] ? String(row[1]).trim() : ""; let ten = row[2] ? String(row[2]).trim() : ""; let soBhxh = row[3] ? String(row[3]).trim() : ""; let soCccd = row[4] ? String(row[4]).trim() : ""; let atv = row[5] ? String(row[5]).trim() : ""; let nndh = row[6] ? String(row[6]).trim() : ""; let namVeHuu = row[7] ? String(row[7]).trim() : ""; let maDonVi = row[8] ? String(row[8]).trim() : ""; let nam = row[9] ? String(row[9]).trim() : ""; let namSinh = row[10] ? String(row[10]).trim() : ""; let nu = row[11] ? String(row[11]).trim() : ""; let chucDanh = row[12] ? String(row[12]).trim() : ""; // Điều kiện bắt dòng nhân sự hợp lệ if (hoTen && hoTen.toLowerCase() !== "họ và tên" && hoTen !== "") { let cleanCccd = soCccd; let cleanMaDonVi = maDonVi.replace(/\s+/g, ''); if (cleanMaDonVi === "") { cleanMaDonVi = "DV-UNKNOWN"; } setMaDonViExcel.add(cleanMaDonVi); maxId++; danhSachUpload.push({ id: maxId, stt_file: stt, ho_ten: hoTen, ten: ten, so_bhxh: soBhxh, so_cccd: cleanCccd, an_toan_vien: atv, nndh: nndh, nam_ve_huu: formatExcelDate(namVeHuu), ma_don_vi: cleanMaDonVi, nam_khao_sat: formatExcelDate(nam), nam_sinh: formatExcelDate(namSinh), gioi_tinh_nu: nu, chuc_danh_cong_viec: chucDanh }); } } if(danhSachUpload.length === 0) { alert("🚨 Không tìm thấy bất kỳ nhân sự nào hợp lệ trong file Excel!"); input.value = ""; return; } if (setMaDonViExcel.size > 0) { let mangMaDonVi = Array.from(setMaDonViExcel); let payloadDonVi = mangMaDonVi.map(m => ({ ma_don_vi: m, ten_don_vi: m === "DV-UNKNOWN" ? "Đơn vị chưa xác định" : `Đơn vị mã ${m}` })); const { error: errDonVi } = await client.from('don_vi').upsert(payloadDonVi, { onConflict: 'ma_don_vi' }); if(errDonVi) console.error("Lỗi đăng ký đơn vị:", errDonVi); } const { error: errNhanSu } = await client.from('nhan_su').insert(danhSachUpload); if(errNhanSu) { console.error("Lỗi Supabase:", errNhanSu); alert("🚨 Lỗi đẩy lên cơ sở dữ liệu: \n" + errNhanSu.message); } else { alert(`🎉 Thành công dứt điểm! Đã đồng bộ thành công ${danhSachUpload.length} nhân sự chuẩn lên Cloud.`); } document.getElementById('search-ns').value = ""; input.value = ""; await taiDuLieuTuCloud(false); } catch (err) { alert("Lỗi cấu trúc tệp Excel: " + err.message); } }; reader.readAsArrayBuffer(file); }
