function napExcelNhanSu(input) { const file = input.files[0]; if(!file) return; const reader = new FileReader(); reader.onload = async function(e) { try { const data = new Uint8Array(e.target.result); const workbook = XLSX.read(data, { type: 'array' }); const firstSheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[firstSheetName]; // Chuyển đổi thành mảng JSON dữ liệu thô const jsonRows = XLSX.utils.sheet_to_json(worksheet); if (!jsonRows || jsonRows.length === 0) { alert("File Excel này trống rỗng, không có dữ liệu dòng nào!"); return; } // Tiến hành quét tìm và ánh xạ động (Tránh lỗi lệch ký tự tiêu đề) const mangNhanSuMapping = jsonRows.map(row => { // Tạo một bản sao chữ thường của toàn bộ key để dò cho dễ const lowerRow = {}; for (let key in row) { lowerRow[key.toString().toLowerCase().trim()] = row[key]; } // Dò tìm cột Họ Tên const ho_ten = row['Họ & Tên'] || row['ho_ten'] || row['Họ và Tên'] || row['Họ Tên'] || lowerRow['ho_ten'] || lowerRow['ho và ten'] || lowerRow['ho ten'] || lowerRow['name'] || ''; // Dò tìm cột Giới Tính const gioi_tinh = row['Giới Tính'] || row['gioi_tinh'] || lowerRow['gioi tinh'] || lowerRow['gioitinh'] || ''; // Dò tìm cột Năm Sinh const nam_sinh = parseInt(row['Năm Sinh'] || row['nam_sinh'] || lowerRow['nam sinh'] || lowerRow['namsinh']) || null; // Dò tìm cột Năm Nghỉ Hưu const nam_ve_huu = parseInt(row['Năm Nghỉ Hưu'] || row['nam_ve_huu'] || lowerRow['nam nghi huu'] || lowerRow['namvehuu']) || null; // Dò tìm cột Mã Đơn Vị const ma_don_vi = String(row['Mã ĐV'] || row['ma_don_vi'] || row['Mã Đơn Vị'] || lowerRow['ma dv'] || lowerRow['ma don vi'] || lowerRow['madv'] || '').trim(); // Dò tìm cột Chi Tiết / Chức Danh const chuc_danh_cong_viec = row['Chi Tiết'] || row['chuc_danh_cong_viec'] || row['Chức Danh'] || lowerRow['chi tiet'] || lowerRow['chuc danh'] || ''; return { ho_ten, gioi_tinh, nam_sinh, nam_ve_huu, ma_don_vi, chuc_danh_cong_viec }; }).filter(x => x.ho_ten !== ''); // Giữ lại những dòng có Tên hợp lệ if(mangNhanSuMapping.length === 0) { alert("Hệ thống không tìm thấy cột nào chứa dữ liệu 'Họ & Tên' hoặc 'Mã ĐV' trong file Excel này. Vui lòng kiểm tra lại dòng tiêu đề đầu tiên!"); return; } alert(`Đang tiến hành đẩy dữ liệu ${mangNhanSuMapping.length} nhân sự lên đám mây...`); const { error } = await client.from('nhan_su').insert(mangNhanSuMapping); if(error) throw error; alert("🎉 Nạp dữ liệu Excel lên Cloud thành công 100%!"); taiDuLieuTuCloud(); } catch (err) { alert("Lỗi kết nối hoặc cấu trúc file: " + err.message); console.error(err); } }; reader.readAsArrayBuffer(file); input.value = ''; // Xóa trạng thái input để lần sau nạp lại file cũ vẫn ăn }
